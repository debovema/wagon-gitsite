38  /**
39   * Handle git check-out and pull from remote site.
40   *
41   * <p>Based on GitCheckOutCommand by Mark Struberg.</p>
42   *
43   * @author Kathryn Huxtable
44   * @see    org.apache.maven.scm.provider.git.gitexe.command.checkout.GitCheckOutCommand
45   */
46  public class GitSiteCheckOutCommand extends AbstractCheckOutCommand implements GitCommand {
47  
48      /**
49       * @see org.apache.maven.scm.command.checkout.AbstractCheckOutCommand#executeCheckOutCommand(org.apache.maven.scm.provider.ScmProviderRepository,
50       *      org.apache.maven.scm.ScmFileSet, org.apache.maven.scm.ScmVersion,
51       *      boolean)
52       */
53      protected CheckOutScmResult executeCheckOutCommand(ScmProviderRepository repo, ScmFileSet fileSet, ScmVersion version,
54              boolean recursive) throws ScmException {
55          GitScmProviderRepository repository = (GitScmProviderRepository) repo;
56  
57          if (GitScmProviderRepository.PROTOCOL_FILE.equals(repository.getFetchInfo().getProtocol())
58                  && repository.getFetchInfo().getPath().indexOf(fileSet.getBasedir().getPath()) >= 0) {
59              throw new ScmException("remote repository must not be the working directory");
60          }
61  
62          int exitCode;
63  
64          CommandLineUtils.StringStreamConsumer stdout = new CommandLineUtils.StringStreamConsumer();
65          CommandLineUtils.StringStreamConsumer stderr = new CommandLineUtils.StringStreamConsumer();
66  
67          // Create or empty the working directory.
68          if (!fileSet.getBasedir().exists()) {
69              if (!fileSet.getBasedir().mkdir()) {
70                  return new CheckOutScmResult("make directory", "The directory create failed.", "", false);
71              }
72          } else {
73              try {
74                  FileUtils.cleanDirectory(fileSet.getBasedir());
75              } catch (IOException e) {
76                  return new CheckOutScmResult("clean directory", "The directory cleanup failed.", e.getMessage(), false);
77              }
78          }
79  
80          Commandline cl = null;
81  
82          // Initialize a new git repo.
83          cl       = createInitCommand(fileSet.getBasedir());
84          exitCode = GitCommandLineUtils.execute(cl, stdout, stderr, getLogger());
85          if (exitCode != 0 || !new File(fileSet.getBasedir(), ".git").exists()) {
86              return new CheckOutScmResult(cl.toString(), "The git-init command failed.", stderr.getOutput(), false);
87          }
88  
89          // Add the remote origin to the git repo.
90          cl       = createRemoteAddOriginCommand(fileSet.getBasedir(), repository);
91          exitCode = GitCommandLineUtils.execute(cl, stdout, stderr, getLogger());
92          if (exitCode != 0) {
93              return new CheckOutScmResult(cl.toString(), "The git-remote command failed.", stderr.getOutput(), false);
94          }
95  
96          // Pull the site branch into master, which checks it out.
97          cl       = createPullCommand(fileSet.getBasedir(), version);
98          exitCode = GitCommandLineUtils.execute(cl, stdout, stderr, getLogger());
99          if (exitCode != 0) {
100             return new CheckOutScmResult(cl.toString(), "The git-pull command failed.", stderr.getOutput(), false);
101         }
102 
103         // And now search for the files.
104         GitListConsumer listConsumer = new GitListConsumer(getLogger(), fileSet.getBasedir(), ScmFileStatus.CHECKED_IN);
105 
106         cl       = GitListCommand.createCommandLine(repository, fileSet.getBasedir());
107         exitCode = GitCommandLineUtils.execute(cl, listConsumer, stderr, getLogger());
108         if (exitCode != 0) {
109             return new CheckOutScmResult(cl.toString(), "The git-ls-files command failed.", stderr.getOutput(), false);
110         }
111 
112         return new CheckOutScmResult(cl.toString(), listConsumer.getListedFiles());
113     }
114 
115     /**
116      * Create a "git init" command.
117      *
118      * @param  workingDirectory the working directory.
119      *
120      * @return the command line to init the local repository.
121      */
122     private Commandline createInitCommand(File workingDirectory) {
123         Commandline cl = GitCommandLineUtils.getBaseGitCommandLine(workingDirectory, "init");
124 
125         return cl;
126     }
127 
128     /**
129      * Create a "git remote add origin" command.
130      *
131      * @param  workingDirectory the working directory.
132      * @param  repository       the SCM repository.
133      *
134      * @return the command line to add the remote origin.
135      */
136     private Commandline createRemoteAddOriginCommand(File workingDirectory, GitScmProviderRepository repository) {
137         Commandline cl = GitCommandLineUtils.getBaseGitCommandLine(workingDirectory, "remote");
138 
139         cl.createArg().setValue("add");
140         cl.createArg().setValue("origin");
141 
142         cl.createArg().setValue(repository.getFetchUrl());
143 
144         return cl;
145     }
146 
147     /**
148      * Create a "git pull origin refs/heads/branch" command.
149      *
150      * @param  workingDirectory the working directory.
151      * @param  version          the remote site branch to check out.
152      *
153      * @return the command line to pull the site branch into the working
154      *         directory.
155      */
156     private Commandline createPullCommand(File workingDirectory, ScmVersion version) {
157         Commandline cl = GitCommandLineUtils.getBaseGitCommandLine(workingDirectory, "pull");
158 
159         cl.createArg().setValue("origin");
160 
161         cl.createArg().setValue("refs/heads/" + version.getName());
162 
163         return cl;
164     }
